Example: 

Ext.onReady(function(){
    var blind = {
        ptype       : 'TDGi.WindowBlind',
        html        : 'Window blind plugin example',
        buttonAlign : 'center',
        height      : 200, // optional
        buttons     : [
            {
                text    : 'hide',
                handler : function() {
                    //this translate to:
                    //this->fbar->blind
                    this.ownerCt.ownerCt.hide();
                }
            }
        ]
    };

    new Ext.Window({
        html    : 'test',
        width   : 400,
        height  : 300,
        plugins : [blind],
        buttons : [
            {
                text : 'Show Blind',
                // This handler runs within the scope of the Ext.Button instance
                handler : function() {
                    //this translate to:
                    //this->fbar->window
                    this.ownerCt.ownerCt.blind.show();
                }
            }
        ]
    }).show();
});
